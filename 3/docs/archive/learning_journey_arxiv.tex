\section{Methodological Commitments (constant across all labs)}
\label{sec:method}
\begin{itemize}
  \item \textbf{From scratch.} Every core algorithm is implemented with only
        \texttt{numpy}/\texttt{matplotlib}/\texttt{scipy}; no black-box
        optimization frameworks, so every line is defensible.
  \item \textbf{Matched-budget baselines.} Each learned/heuristic method is
        compared against a baseline under an \emph{identical} computational
        budget (equal node expansions / constraint checks / fitness
        evaluations).
  \item \textbf{Statistical validation.} Stochastic methods are run over
        multiple seeds; differences are tested (e.g., Wilcoxon signed-rank,
        Friedman) rather than asserted from a single run.
  \item \textbf{Reproducibility.} Fixed seeds, pinned dependencies, scripted
        runners.
\end{itemize}

\section{Lab 3 --- Population-Based / Swarm Optimisation
        (Wi-Fi AP Placement)}
\label{sec:lab3}
This lab is fully realised (the others above are templated). It is reproduced
here in publishable form.

\subsection{Problem}
Place $K$ Wi-Fi access points on a University of Dhaka residential-hall floor
$[0,W]\times[0,H]$ to maximise occupancy-weighted room coverage under a
minimum-separation (co-channel) rule and floor boundaries. Reference instance:
$W\times H = \SI{60}{\metre}\times\SI{40}{\metre}$, $N=40$ jittered rooms with
occupancy weights, three interior concrete partitions, and a tight budget of
$K=3$ APs (search space $\mathbb{R}^{6}$).

\subsection{Method}
A from-scratch inertia-weight Particle Swarm Optimizer~\cite{kennedy1995,
shi1998}. A particle is the flat vector of AP coordinates
$\mathbf{x}=[a^x_0,a^y_0,\dots]\in\mathbb{R}^{2K}$; the velocity update is
\begin{equation}
\mathbf{v}\leftarrow w\,\mathbf{v}
  + c_1 r_1(\mathbf{p}_{\text{best}}-\mathbf{x})
  + c_2 r_2(\mathbf{g}_{\text{best}}-\mathbf{x}),\qquad
\mathbf{x}\leftarrow\mathbf{x}+\mathbf{v},
\end{equation}
with linearly decayed inertia $w:0.9\!\to\!0.4$, $c_1=c_2=1.49445$, and velocity
clamping. The objective is
\begin{equation}
F(\mathbf{x}) = 100\cdot\frac{\sum_i w_i\,\sigma\!\big((\max_j \mathrm{RSSI}_{ij}-\tau)/s\big)}{\sum_i w_i}
  \;-\;\lambda_{\text{sep}}\!\!\sum_{j<k}\big[\max(0,d_{\min}-\lVert\mathbf{a}_j-\mathbf{a}_k\rVert)\big]^2,
\end{equation}
with $\mathrm{RSSI}_{ij}=P_{\text{tx}}-(L_0+10n\log_{10}\max(d_{ij},1))-\gamma c_{ij}$
(log-distance path loss; $c_{ij}$ = wall crossings). Boundaries are handled by
clamping (repair); separation by penalty.

\subsection{Experimental setup}
Swarm $P=30$, iterations $T=100$ (budget $P(T{+}1)=3030$ fitness evaluations),
15 independent seeds. Baselines at the \emph{identical} budget: Random Search
(3030 samples) and an exhaustive coarse Grid Search
($\binom{25}{3}=2300$ site combinations).

\subsection{Results}
\begin{table}[h]\centering
\caption{Wi-Fi placement: PSO vs matched-budget baselines (weighted coverage
\%; mean over 15 runs).}
\begin{tabular}{lccc}
\toprule
Method & Evals/run & Mean fitness & Std \\
\midrule
PSO (ours)   & 3030 & \textbf{89.91} & 0.26 \\
Random Search& 3030 & 87.43 & 0.43 \\
Grid Search  & 2300 & 89.46 & 0.00 \\
\bottomrule
\end{tabular}
\end{table}

PSO beats Random by \SI{2.83}{\percent} weighted coverage; the advantage is
significant under a one-sided Wilcoxon signed-rank test ($p=\num{3.05e-5}$). PSO
also exceeds the exhaustive coarse grid because the jittered optimum lies
off-grid — a continuous-search advantage. See \autoref{fig:lab3}.

\caption{Left: convergence (PSO overtakes both baselines). Right: floor coverage
heatmap with the swarm global-best trajectory settling into a spread layout.

\label{fig:lab3}
\end{figure}

\subsection{What was discovered}
\begin{itemize}
  \item Under a tight AP budget, \emph{placement} dominates: random layouts
        strand the far/ shadowed rooms; smart spread recovers $\sim$2.8 points.
  \item PSO's variance ($\sigma=0.26$) is far below Random's ($0.43$):
        reliability, not just mean quality.
  \item Grid Search cannot scale — a fine enough grid needs
        $\binom{G}{K}\!\gg\!$ budget — a concrete curse-of-dimensionality lesson.
\end{itemize}
